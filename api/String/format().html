String.prototype.format(Array|Object values, [RegExp pattern=/\{([\w_\-]+)\}/g]) => String

1'Hi {0}, welcome to {1} !'.format(['Lytc', '"One"']))
2// => 'Hi Lytc, welcome to "One" !'
3
4'Hi {name}, welcome to {lib} !'.format({name: 'Lytc', lib: '"One"'}))
5// => 'Hi Lytc, welcome to "One" !'
6
7'Hi [0], welcome to [1] !'.format(['Lytc', '"One"'], /\[([\w_\-]+)\]/g))
8// => 'Hi Lytc, welcome to "One" !'
