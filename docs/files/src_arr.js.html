/**
 * @class arr
 */
(function ($) {
    "use strict"

    /**
     * @method arr
     * @param {Array} arr
     * @return {Array}
     */
    $.arr = function (arr) {
        $.extend(arr, $.arr.fn)
        return arr
    }

    $.arr.fn = {
        /**
         * @method pad
         * @param {Int} size
         * @param {Mixed} value
         * @return {Array}
         */
        pad: function (size, value) {
            if (size > 0) {
                for (var i = 0, len = size - this.length; i < len; ++i) {
                    this.push(value)
                }
            } else {
                for (var i = 0, len = -size - this.length; i < len; ++i) {
                    this.unshift(value)
                }
            }

            return this
        }

        /**
         * @method padLEft
         * @param {Int} size
         * @param {Mixed} value
         * @return {Array}
         */
        ,padLeft: function (size, value) {
            return this.pad(-size, value)
        }

        /**
         * @method uniq
         * @return {Array}
         */
        ,uniq: function () {
            var result = []
            for (var i = 0, len = this.length; i < len; ++i) {
                if (-1 == result.indexOf(this[i])) {
                    result.push(this[i])
                }
            }

            return $.arr(result)
        }

        /**
         * @method truthy
         * @return {Array}
         */
        ,truthy: function () {
            var result = this.filter(function (item) {
                return !!item
            })
            return $.arr(result)
        }

        /**
         * @method falsy
         * @return {Array}
         */
        ,falsy: function () {
            var result = this.filter(function (item) {
                return !item
            })
            return $.arr(result)
        }

        /**
         * @method exclude
         * @param {Mixed|Array} items*
         * @return {Array}
         */
        ,exclude: function () {
            var arg, index
            for (var i = 0, len = arguments.length; i < len; ++i) {
                arg = $.isArray(arguments[i]) ? arguments[i] : [arguments[i]]
                for (var j = 0, jlen = arg.length; j < jlen; ++j) {
                    if (-1 != (index = this.indexOf(arg[j]))) {
                        this.splice(index, 1)
                    }
                }
            }
            return this
        }

        /**
         * @method toArray
         * @return {Array}
         */
        ,toArray: function () {
            var result = []
            for (var i = 0, len = this.length; i < len; ++i) {
                result.push(this[i])
            }
            return result
        }
    }
})(one)
