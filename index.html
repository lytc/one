</body>

Selectors

One.js selector wrap "querySelectorAll" to support more features like: Adjacent previous sibling combinator "-", General previous sibling combinator "<~"

//Adjacent previous sibling combinator
$('#container').find('- div')

//General previous sibling combinator
$('#container').find('<~ div')

Ajax

One.js ajax support full XMLHttpRequest Level 2 and modify a bit to make it more convenient.

$.ajax('/hello', function(response) {
  // do something
})

$.ajax('/hello', {
  onStart: function(e) {
    // do something
  }
  ,onSuccess: function(response, e) {
    // do something
  }
})

$.getJson('/data.json', function(data) {
  // do something
})

JSONP

JSONP is not a part of XMLHttpRequest. So One.js seperate it from ajax.

$.getJsonP('http://data.json', function(data) {
  // do something
})

Template

One.js Template compiles JavaScript templates into functions that can be evaluated for rendering. Useful for rendering complicated bits of HTML from JSON data sources

var tpl = 'Hi <%= this.name %>, welcome to "<%= this.lib %>"!'
var content = $.template(template, {name: 'Lytc', lib: 'One.js'})
=> 'Hi Lytc, welcome to "One.js"!'

var template = $.template(
  "<% this.items.forEach(function(item) { %>",
    "<li><%= item %></li>",
  "<% }) %>"
)
template.render({items : ['foo', 'bar', 'baz']})
=> "<li>foo</li>  <li>bar</li>  <li>baz</li>"
